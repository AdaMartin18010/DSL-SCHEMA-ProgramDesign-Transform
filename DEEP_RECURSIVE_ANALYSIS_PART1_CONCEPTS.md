class CorrectInformationTheory:
    """
    âœ… æ­£ç¡®çš„ä¿¡æ¯ç†µå®ç°

    å‚è€ƒ: Cover & Thomas "Elements of Information Theory"
    """

    def calculate_entropy(self, probabilities: List[float]) -> float:
        """
        è®¡ç®—ç¦»æ•£éšæœºå˜é‡çš„é¦™å†œç†µ

        H(X) = -Î£ p(x) logâ‚‚ p(x)
        """
        entropy = 0.0
        for p in probabilities:
            if p > 0:  # é¿å… log(0)
                entropy -= p * math.log2(p)
        return entropy

    def calculate_schema_entropy(self, schema: Dict) -> float:
        """
        è®¡ç®—Schemaçš„ä¿¡æ¯ç†µ

        æ­¥éª¤:
        1. æå–Schemaçš„æ‰€æœ‰å…ƒç´ ï¼ˆç±»å‹ã€å­—æ®µã€çº¦æŸï¼‰
        2. è®¡ç®—å„å…ƒç´ çš„æ¦‚ç‡åˆ†å¸ƒ
        3. åº”ç”¨é¦™å†œç†µå…¬å¼
        """
        # 1. æå–å…ƒç´ é¢‘ç‡
        elements = self._extract_schema_elements(schema)
        total = sum(elements.values())

        # 2. è®¡ç®—æ¦‚ç‡åˆ†å¸ƒ
        probabilities = [count / total for count in elements.values()]

        # 3. è®¡ç®—ç†µ
        return self.calculate_entropy(probabilities)

    def calculate_mutual_information(self,
                                    joint_probs: Dict[Tuple, float],
                                    marginal_x: Dict,
                                    marginal_y: Dict) -> float:
        """
        è®¡ç®—ä¸¤ä¸ªSchemaçš„äº’ä¿¡æ¯

        I(X;Y) = Î£Î£ P(x,y) log [P(x,y) / (P(x)P(y))]
        """
        mi = 0.0
        for (x, y), p_xy in joint_probs.items():
            p_x = marginal_x.get(x, 0)
            p_y = marginal_y.get(y, 0)

            if p_xy > 0 and p_x > 0 and p_y > 0:
                mi += p_xy * math.log2(p_xy / (p_x * p_y))

        return mi

class CorrectMultimodalFusion:
    """
    âœ… æ­£ç¡®çš„å¤šæ¨¡æ€èåˆå®ç°

    å‚è€ƒ: CLIP (OpenAI), KGA (ä¸œå—å¤§å­¦ 2025)

    def __init__(self):
        # åŠ è½½CLIPæ¨¡å‹ç”¨äºè·¨æ¨¡æ€å¯¹é½
        from transformers import CLIPModel, CLIPProcessor
        self.clip = CLIPModel.from_pretrained("openai/clip-vit-large-patch14")
        self.processor = CLIPProcessor.from_pretrained("openai/clip-vit-large-patch14")

        # å¯å­¦ä¹ çš„é—¨æ§ç½‘ç»œ
        self.gate_network = nn.Sequential(
            nn.Linear(512 * 2, 256),
            nn.ReLU(),
            nn.Linear(256, 2),
            nn.Softmax(dim=-1)
        )

        # è·¨æ¨¡æ€æ³¨æ„åŠ›
        self.cross_attention = nn.MultiheadAttention(
            embed_dim=512, num_heads=8
        )

    def encode_text(self, text: str) -> torch.Tensor:
        """ç¼–ç æ–‡æœ¬åˆ°ç»Ÿä¸€ç©ºé—´"""
        inputs = self.processor(text=[text], return_tensors="pt", padding=True)
        text_features = self.clip.get_text_features(**inputs)
        return F.normalize(text_features, p=2, dim=-1)

    def encode_image(self, image: Image.Image) -> torch.Tensor:
        """ç¼–ç å›¾åƒåˆ°ç»Ÿä¸€ç©ºé—´"""
        inputs = self.processor(images=image, return_tensors="pt")
        image_features = self.clip.get_image_features(**inputs)
        return F.normalize(image_features, p=2, dim=-1)

    def fuse(self, text: str, image: Image.Image) -> torch.Tensor:
        """
        çœŸæ­£çš„å¤šæ¨¡æ€èåˆ

        æ­¥éª¤:
        1. ç¼–ç åˆ°ç»Ÿä¸€ç©ºé—´ï¼ˆCLIPå¯¹é½ï¼‰
        2. è®¡ç®—è‡ªé€‚åº”é—¨æ§æƒé‡
        3. åº”ç”¨è·¨æ¨¡æ€æ³¨æ„åŠ›
        4. è¿”å›èåˆè¡¨ç¤º
        """
        # 1. ç¼–ç 
        text_emb = self.encode_text(text)
        image_emb = self.encode_image(image)

        # 2. é—¨æ§æƒé‡
        concat = torch.cat([text_emb, image_emb], dim=-1)
        gate_weights = self.gate_network(concat)

        # 3. åŠ æƒ
        weighted_text = gate_weights[0, 0] * text_emb
        weighted_image = gate_weights[0, 1] * image_emb

        # 4. è·¨æ¨¡æ€æ³¨æ„åŠ›

            query=weighted_text.unsqueeze(0),
            key=weighted_image.unsqueeze(0),
            value=weighted_image.unsqueeze(0)
        )

        return fused.squeeze(0), attention_weights
